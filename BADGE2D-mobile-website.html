Quick Start (JS)

JAVASCRIPT 
// 1. Create engine + scene
const engine = new Engine(canvas);
const scene  = new Scene('Level1');
scene.load(new Vector2(0, 900));

// 2. Create an entity
const e = scene.createEntity('Player', 0, 0);
const sp = new Sprite();
sp.w = 32; sp.h = 32;
sp.color = Color.fromHex('#ff6b00');
scene.world.add(e, sp);

// 3. Update loop
engine.onUpdate = (dt) => {
  const ax = engine.input.getAxis(
    'ArrowLeft', 'ArrowRight');
  // move player...
};

engine.loadScene(scene);
engine.start();

ECS World

JAVASCRIPT 
const world = new World();

// Add components
const e = world.create('Box');
world.add(e, new Transform(100, 200));
world.add(e, new Sprite());

// Query — generator, zero alloc
for (const [e, tr, sp]
  of world.each(Transform, Sprite)) {
  tr.x += 1; // move right
}

// Destroy
world.destroy(e);

Physics

JAVASCRIPT 
const rb = new Rigidbody('dynamic');
rb.w = 40; rb.h = 40;
rb.restitution = 0.5;

scene.addPhysicsBody(e, rb);

// Apply impulse
rb.vx = 200; rb.vy = -400;

// Collision callback
scene.physWorld.onCollision
  .push((ea, eb) => {
    console.log('hit!', ea, eb);
  });
